Which Number is Greater? 

a = int(input('What is your first number?: '))
b = int(input('What is your second number?: '))

if a>b:
    print('{} is greater than {}'.format(a,b))
elif a==b:
    print('{} is the same as {}'.format(a,b))
else:
    print('{} is greater than {}'.format(b,a))
